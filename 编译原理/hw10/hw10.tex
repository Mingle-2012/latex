\subsection*{1. 下面是一段 C 程序，请写出对应生成的三地址代码。}

\begin{lstlisting}[language=C]
{
    int i = 0;
    int a[10];
    while (i <= 9) {
        a[i] = i;
        i++;
    }
}
\end{lstlisting}

\noindent\textbf{{\heiti 解答}}\\ 

\begin{lstlisting}
mov  i,,0
le   t1,i,9
jmpf t1,,10
mov  t2,,c    ; where c is base_a - 0 * 4
mult t3,i,4
mov  t2[t3],,i
add  t4,i,1
mov  i,,t4
jmp  ,,2
              ;
\end{lstlisting}
